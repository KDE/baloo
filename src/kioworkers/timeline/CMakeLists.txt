)

    IDENTIFIER Baloo::KIO_TIMELINE

    OLD_CATEGORY_NAMES kf5.kio.kio_timeline kf.kio.slaves.timeline

  KF5::KIOCore
  KF5::I18n
  KF5::Baloo
)
